SELECT --> Оператор запроса данных из таблицы.

SELECT * --> Извлечь всё.

SELECT name --> Извлечь определённые столбцы (name это название столбца).

SELECT name AS name_1 --> Выдать столбцу псевдоним (name заменить на name_1).

SELECT DISTINCT (name) 
FROM name_table --> Извлекает уникальные значения из столбца name и таблицы
name_table.

FROM --> Оператор ставитсяперед названием таблицы откуда будем извлекать 
данные.

DISTINCT --> Извлекает уникальные значения в столбце.

LIMIT --> Ограничение количества извлекаемых строк в результате.

AS --> Создание псевдонима.
